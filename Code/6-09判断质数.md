### 时间复杂度O(n)
```js
  function  isPrime(num){
      for (var i = 2;i <= n-1;i++) {
        if(num % i ==0){
          return false;
        }
      }
      return true;
    }
```
### 时间复杂度O(logN)
```js
  function  isPrime(num){
      var temp = parseInt(Math.sqrt(num));
      for (var i = 2;i <= temp;i++) {
        if(num % i ==0){
          return false;
        }
      }
      return true;
    }
```
